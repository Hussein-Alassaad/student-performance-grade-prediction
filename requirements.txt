        for (int j = 0; j < arr.length; j++) {
            if (i != j && arr[i] == arr[j]) {
                return true; 
                }}}

✅ PRO WAY: 
public static boolean hasDuplicates(int[] arr) {
    Set<Integer> seen = new HashSet<>();
    for (int num : arr) {
        if (!seen.add(num)) return true;
    }
    return false;
}
